Using CSS Counters

body {
counter-reset: section;
}
h2::before {
counter-increment: section;
content: "Section-" counter(section) ": ";
}

ul {
counter-reset: count;
list-style-type: none;
}
li::before {
counter-increment: count;
content: counters(count, ".") " ";
}
div {
border: 2px solid red;
padding: 20px;
margin: 20px;
}
