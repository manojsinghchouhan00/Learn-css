SELECTOR EXAMPLE THEORY
[attributeName] a[target] Selects all a elements with a target Attribute
[attributeName=value] a[target="_blank"] Selects all a elements with target="_blank"
[attributeName~=value] [title~="flower"] Selects all elements with a title attribute that contains a space-separated words, one of which is "flower"
[attributeName|=value] [class|="first-page"] Selects all elements with a class attributeName value (first-page) starting with "first" the specified value followed by a hyphen (-)
[attributeName^=value] [class^="first"] Selects all elements with a class attributeName value starting with "first" (like-"first-page,firstPage")
[attributeName$=value] [class$="test"] Selects all elements with a class attributeName value ending with "test"
[attributeName*=value] [class*="te"] Selects all elements with a class attributeName value containing the substring "te"
